# 1、确认安装node.js和yarn

确保Node.js的版本 >= 8

# 2、安装vuepress

yarn global add vuepress

# 3、新建项目

# 新建项目
mkdir xiaojunblog

# 进入目录 创建 docs目录
cd xiaojunblog
mkdir docs

# 新建一个Markdown文件
echo '# hello xiaojun' > docs/README.md

# 本地预览
npx vuepress dev docs

# 4、新建配置文件

# 配置文件名 package.json 和docs同目录
npm init -y 或者 yarn init -y

# 5、编辑配置文件并执行命令生成静态的HTMl文件

{
   "scripts": {
     "docs:dev": "vuepress dev docs",
     "docs:build": "vuepress build docs"
   }
}

要生成静态的HTML文件 命令如下：

yarn docs:build

默认情况下 文件将会被生成在.vuepress/dist下面，也可以通过.vuepress/config.js中的dest字段修改

# 7、在config.js文件中配置网站标题、描述等信息

module.exports = {
    title: "xiaojun's blog",
    description: "这是相当于HTML中meta的描述信息 SEO会用到",
    head: [
        // 增加一个自定义的 favicon(网页标签的图标)
        // 这里的 '/' 指向 docs/.vuepress/public 文件目录 
        // 即 docs/.vuepress/public/logo.png
        ['link', { rel: 'icon', href: '/logo.png'}],
    ],
    base: '/', //路由的路径
    markdowm: {
        lineNumbers: false //代码块是否显示行号
    },
    //主题配置
    themeConfig: {
        //导航栏配置
        nav: [
            //单项的
            {text: '开发知识', link: '/technology/'},
            {text: '生活随笔', link: '/life/'},
            //多项的(可以下拉)
            {
                text: '其他',
                items: [
                    { text : '微博', link: 'https://www.weibo.com'},
                    { text : '知乎', link: 'https://www.zhihu.com'},
                ]
            },
        ],
        //侧边栏配置
        sidebar: {
            // 打开technology主页链接时生成下面这个菜单
            '/technology/': [
                {
                    title: 'iOS',
                    children: [
                        // ['路径', '标题']
                        // 或者写成 '路径',标题自动识别为该地址的文件中的h1标题
                        // 不以 '/' 结尾的就是指向.md文件
                        ['/technology/ios/' , '测试']
                    ]
                },
                {
                    title: 'Mac',
                    children: [
                        // ['路径', '标题']
                        // 或者写成 '路径',标题自动识别为该地址的文件中的h1标题
                        // 不以 '/' 结尾的就是指向.md文件
                        ['/technology/mac/' , '测试']
                    ]
                },
                ['/others/', '其他']
            ]
        },
        sidebarDepth: 2, //侧边栏显示2级
        //
    }

# 注意事项

  • 和图标/图片等静态资源相关的 其 '/' 默认指向的是 docs/.vuepress/public/
  • 和侧边栏/导航栏相关的地址配置 其 '/' 默认指向的是 docs/
  • 侧边栏/导航栏指向.md文件的需要先建立相关.md文件，不然会报404或者页面空白

# 9、部署

  • 在项目根目录 细节脚本文件

    cd xiaojunblog
touch deploy.sh

  • 脚本文件内容

    #!/usr/bin/env sh

#确保脚本抛出遇到的错误
set -e

# 生成静态文件
npm run docs:build

# 进入生成的文件夹
cd docs/.vuepress/dist

# 如果是发布到自定义域名
# echo 'www.example.com' > CNAME

git init
git add -A
git commit -m "deploy"
